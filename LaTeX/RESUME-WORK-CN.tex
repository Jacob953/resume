func permute(nums []int) [][]int {
	res := make([][]int, 0)
	var dfs func(in, out []int)
	dfs = func(in, out []int) {
		if len(in) == 0 {
			res = append(res, out)
			return
		}
		for idx := range in {
			add := append(out, in[idx])
			left := in[:idx]
			if idx < len(in)-1 {
				left = append(left, in[idx+1:]...)
			}
			dfs(left, add)
		}
	}
	dfs(nums, []int{})
	return res
}
